# 课程

#获得用户课程信息
method:get
paths:/course
consumes:application/json/
request:{
    profile_name:''(用户档案名称,必写)
    tag:'ain'(获取这个用户自己的)or'follower'（获取这个用户加入别人的）
}
return：{
    courses:{},
}
#添加一个课程
method:post
paths:/course
consumes:application/json
request:{
    code:'xxx'(课程code),
    name：''（课程名称）,
    course_type:''(课程类型)
}

return：{
    status:True(False),
    msg:''
}
